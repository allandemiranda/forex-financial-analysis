48 void Chart::setTimeChart(std::string* time) {
49  std::vector<std::string> tempos_permitidos = {
50  "M1", "M2", "M3", "M4", "M5", "M6", "M10", "M12", "M15", "M20", "M20",
51  "H1", "H2", "H3", "H4", "H6", "H8", "H12", "D1", "W1", "J1", "Y1"};
52  // OBS: J1 = Mês 1
53  // 1 hora = 3600 segundos
54  // 1 dia = 86400 segundos
55  // 1 semana = 604800 segundos
56  // 1 mês (30dias) = 2592000 segundos
57  // 1 ano (365dias) = 31536000 segundos
58  unsigned long time_final = 0;
59 #pragma omp parallel
60  {
61 #pragma omp for
62  for (unsigned long i = 0; i < tempos_permitidos.size(); ++i) {
63  if (*time == tempos_permitidos[i]) {
64  if (time->at(0) == 'M') {
65 #pragma omp critical
66  {
67  time->erase(0, 1);
68  time_final = (unsigned long)(60 * (unsigned long)std::stoi(*time));
69  }
70 #pragma omp cancel for
71  }
72  if (time->at(0) == 'H') {
73 #pragma omp critical
74  {
75  time->erase(0, 1);
76  time_final =
77  (unsigned long)(3600 * (unsigned long)std::stoi(*time));
78  }
79 #pragma omp cancel for
80  }
81  if (time->at(0) == 'D') {
82 #pragma omp critical
83  {
84  time->erase(0, 1);
85  time_final =
86  (unsigned long)(86400 * (unsigned long)std::stoi(*time));
87  }
88 #pragma omp cancel for
89  }
90  if (time->at(0) == 'D') {
91 #pragma omp critical
92  {
93  time->erase(0, 1);
94  time_final =
95  (unsigned long)(86400 * (unsigned long)std::stoi(*time));
96  }
97 #pragma omp cancel for
98  }
99  if (time->at(0) == 'J') {
100 #pragma omp critical
101  {
102  time->erase(0, 1);
103  time_final =
104  (unsigned long)(2592000 * (unsigned long)std::stoi(*time));
105  }
106 #pragma omp cancel for
107  }

332 void Chart::convertingToTimeVector(time_t* data_inicial, time_t* data_final) {
333  struct date_time {
334  time_t inicial;
335  time_t final;
336  };
337  std::vector<date_time> lista_de_intervalos;
338  while (*data_inicial < *data_final) {
339  date_time nova_date_time;
340  nova_date_time.inicial = *data_inicial;
341  *data_inicial += *getTimeChart();
342  nova_date_time.final = *data_inicial;
343  lista_de_intervalos.push_back(nova_date_time);
344  lista_de_intervalos.shrink_to_fit();
345  }
346  std::vector<Candlestick> novo_chart;
347 #pragma omp parallel
348  {
349 #pragma omp for
350  for (unsigned long i = 0; i < lista_de_intervalos.size(); ++i) {
351  auto novo_vt = candleSearch(&chart, lista_de_intervalos[i].inicial);
352  if (chart.end() == novo_vt) {
353  time_t minuto = 60;
354  Candlestick temporario_(&minuto, &lista_de_intervalos[i].inicial);
355  lista_de_intervalos[i].inicial += minuto;
356  auto vt = candleSearch(&chart, lista_de_intervalos[i].inicial);
357  while (lista_de_intervalos[i].inicial < lista_de_intervalos[i].final) {
358  if (chart.end() == vt) {
359  Candlestick temporario_dois(&minuto,
360  &lista_de_intervalos[i].inicial);
361  temporario_ = temporario_ + temporario_dois;
362  lista_de_intervalos[i].inicial += minuto;
363  } else {
364  temporario_ = temporario_ + *vt;
365  lista_de_intervalos[i].inicial += *vt->getTime();
366  }
367  std::advance(vt, 1);
368  if (vt == chart.end()) {
369  if (*temporario_.getTime() < *getTimeChart()) {
370  temporario_.addTime(*getTimeChart() - *temporario_.getTime());
371  }
372  break;
373  }
374  if (*vt->getDate() != lista_de_intervalos[i].inicial) {
375  if (*vt->getDate() >= lista_de_intervalos[i].final) {
376  if (*temporario_.getTime() < *getTimeChart()) {
377  temporario_.addTime(*getTimeChart() - *temporario_.getTime());
378  }
379  break;
380  } else {
381  vt = chart.end();
382  }
383  }
384  }
385 #pragma omp critical
386  { novo_chart.push_back(temporario_); }
387  } else {
388  time_t minuto = 60;
389  lista_de_intervalos[i].inicial += minuto;
390  auto vt = candleSearch(&chart, lista_de_intervalos[i].inicial);
391  while (lista_de_intervalos[i].inicial < lista_de_intervalos[i].final) {
392  if (chart.end() == vt) {
393  Candlestick temporario_dois(&minuto,
394  &lista_de_intervalos[i].inicial);
395  *novo_vt = *novo_vt + temporario_dois;
396  lista_de_intervalos[i].inicial += minuto;
397  } else {
398  *novo_vt = *novo_vt + *vt;
